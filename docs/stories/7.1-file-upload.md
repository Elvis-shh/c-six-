## 一、PRD（产品需求文档）

### 1.1 背景

部分用户的财报数据可能不在系统预置范围内，需要支持上传本地财报文件（PDF/Word/Excel/图片/TXT）进行解析和分析。

### 1.2 需求描述

作为用户，我可以上传 PDF/Word/Excel/图片/TXT 格式的财报文件，系统自动解析并生成分析报告。

### 1.3 验收标准

| # | 验收条件 |
|---|---------|
| AC1 | 搜索页面 + Dashboard Navbar 均支持文件上传入口 |
| AC2 | 支持点击上传、拖拽上传、粘贴上传（Ctrl+V） |
| AC3 | 文件类型校验：PDF/DOC/DOCX/XLS/XLSX/TXT/JPG/PNG |
| AC4 | 单文件大小限制 ≤ 50MB，超限提示 |
| AC5 | 上传后显示进度条 |
| AC6 | 上传完成后自动跳转处理流程（OCR→NER→Dashboard） |

@Service
public class MinioService {
    private final MinioClient client;

    public String upload(InputStream stream, String objectName, String contentType, long size) {
        client.putObject(PutObjectArgs.builder()
            .bucket(bucket)
            .object(objectName)
            .stream(stream, size, -1)
            .contentType(contentType)
            .build());
        return String.format("%s/%s/%s", endpoint, bucket, objectName);
    }

    public InputStreamResource download(String objectName) { ... }
    public void delete(String objectName) { ... }
}

**步骤 2：UploadController**

```java
@PostMapping("/upload/report")
public ApiResponse<UploadTaskResponse> upload(
    @RequestParam("file") MultipartFile file) {

    // 1. 类型校验
    fileValidationService.validate(file);

    // 2. 上传到 MinIO
    String objectName = generateObjectName(file.getOriginalFilename());
    String fileUrl = minioService.upload(file.getInputStream(), objectName,
        file.getContentType(), file.getSize());

    // 3. 创建任务记录
    String taskId = UUID.randomUUID().toString();
    uploadTaskService.createTask(taskId, file.getOriginalFilename(), fileUrl);

    // 4. 异步触发 OCR 解析
    uploadTaskService.processAsync(taskId);

    return ApiResponse.accepted(UploadTaskResponse.builder()
        .taskId(taskId)
        .status("pending")
        .fileName(file.getOriginalFilename())
        .message("文件已上传，正在排队解析...")
        .build());
}
```

**步骤 3：FileUpload.vue**

```vue
<script setup lang="ts">
const emit = defineEmits<{ uploaded: [taskId: string] }>()

const dragging = ref(false)
const uploading = ref(false)
const progress = ref(0)
const error = ref('')

const ACCEPTED_TYPES = '.pdf,.doc,.docx,.xls,.xlsx,.txt,.jpg,.jpeg,.png'
const MAX_SIZE = 50 * 1024 * 1024

function validateFile(file: File): string | null {
  const ext = file.name.split('.').pop()?.toLowerCase()
  const allowed = ['pdf','doc','docx','xls','xlsx','txt','jpg','jpeg','png']
  if (!allowed.includes(ext || '')) return '不支持的文件类型'
  if (file.size > MAX_SIZE) return '文件大小不能超过 50MB'
  return null
}

async function uploadFile(file: File) {
  const err = validateFile(file)
  if (err) { error.value = err; return }

  uploading.value = true
  progress.value = 0

  const formData = new FormData()
  formData.append('file', file)

  try {
    const res = await api.post('/api/v1/upload/report', formData, {
      onUploadProgress: (e) => {
        if (e.total) progress.value = Math.round((e.loaded / e.total) * 100)
      }
    })
    emit('uploaded', res.data.data.taskId)
  } catch (e) {
    error.value = '上传失败，请重试'
  } finally {
    uploading.value = false
  }
}

// 拖拽事件
function onDrop(e: DragEvent) {
  dragging.value = false
  const file = e.dataTransfer?.files[0]
  if (file) uploadFile(file)
}

// 粘贴事件
function onPaste(e: ClipboardEvent) {
  const file = e.clipboardData?.files[0]
  if (file) uploadFile(file)
}
</script>

<template>
  <div
    class="upload-zone"
    :class="{ dragging }"
    @dragover.prevent="dragging = true"
    @dragleave="dragging = false"
    @drop.prevent="onDrop"
    @paste="onPaste"
  >
    <div v-if="!uploading">
      <div class="upload-icon">📁</div>
      <p>拖拽文件到此处，或 <label class="upload-link">点击上传<input type="file" hidden :accept="ACCEPTED_TYPES" @change="onFileChange" /></label></p>
      <p class="upload-hint">支持 PDF / Word / Excel / TXT / 图片，单文件最大 50MB</p>
    </div>
    <div v-else class="upload-progress">
      <div class="progress-bar">
        <div class="progress-fill" :style="{ width: progress + '%' }" />
      </div>
      <p>上传中... {{ progress }}%</p>
    </div>
    <p v-if="error" class="upload-error">{{ error }}</p>
  </div>
</template>
```
