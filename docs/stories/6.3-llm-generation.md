# Story 6.3 — LLM 对话生成

> **Epic**: chat (智能问答) | **优先级**: P2 | **阶段**: Phase 3 AI 增强
> **依赖**: Story 6.2 | **预估工时**: 3d

---

## 一、PRD（产品需求文档）

### 1.1 背景

RAG 检索提供了上下文，LLM 负责基于上下文生成自然语言回答。需要支持流式输出（SSE）以降低用户等待感知，并支持多 Provider 热备。

### 1.2 需求描述

作为系统，我需要调用 LLM 生成财报分析回答，支持流式输出和多 LLM Provider 切换。

### 1.3 验收标准

| # | 验收条件 |
|---|---------|
| AC1 | 用户提问后，助手回复逐字显示（打字机效果） |
| AC2 | 回答基于 RAG 检索结果，引用具体财报数据 |
| AC3 | 回答末尾附带参考来源标注（refs） |
| AC4 | 支持 DeepSeek/OpenAI/智谱 多 Provider 切换 |
| AC5 | LLM 调用超时 30s 后有兜底回复 |
| AC6 | Token 用量被记录和统计 |

### 1.4 数据流

```
用户提问 → Java ChatController
  → Java ChatService
    → Python POST /ai/v1/rag/search (获取 RAG Context)
    → Python POST /ai/v1/chat/generate (LLM SSE 流式生成)
    → Java 转发 SSE 到前端
    → 前端逐 token 渲染
```

**步骤 2：LLM 多 Provider 服务**

```python
# app/services/llm_service.py
import httpx
import json
from typing import AsyncGenerator

class LLMService:
    PROVIDERS = {
        "deepseek": {
            "url": "https://api.deepseek.com/v1/chat/completions",
            "model": "deepseek-chat",
        },
        "openai": {
            "url": "https://api.openai.com/v1/chat/completions",
            "model": "gpt-4o",
        },
        "zhipu": {
            "url": "https://open.bigmodel.cn/api/paas/v4/chat/completions",
            "model": "glm-4",
        },
    }

    def __init__(self, provider="deepseek", api_key=None):
        self.provider = provider
        self.api_key = api_key
        self.config = self.PROVIDERS[provider]

    async def chat_stream(
        self, messages: list[dict], temperature=0.3, max_tokens=2048
    ) -> AsyncGenerator[str, None]:
        """SSE 流式生成"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json",
        }
        payload = {
            "model": self.config["model"],
            "messages": messages,
            "temperature": temperature,
            "max_tokens": max_tokens,
            "stream": True,
        }
        async with httpx.AsyncClient(timeout=30) as client:
            async with client.stream("POST", self.config["url"],
                                      json=payload, headers=headers) as response:
                async for line in response.aiter_lines():
                    if line.startswith("data: "):
                        data = line[6:]
                        if data == "[DONE]":
                            break
                        try:
                            chunk = json.loads(data)
                            delta = chunk["choices"][0].get("delta", {})
                            content = delta.get("content", "")
                            if content:
                                yield content
                        except (json.JSONDecodeError, KeyError):
                            continue

    def build_messages(self, system_prompt: str, user_message: str,
                       rag_contexts: list[dict]) -> list[dict]:
        """构建消息列表"""
        # 拼接 RAG 上下文
        rag_text = "\n\n".join([
            f"[来源: {c['source']}]\n{c['content']}"
            for c in rag_contexts
        ])
        reference = REFERENCE_TEMPLATE.format(rag_context=rag_text)

        return [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": f"{reference}\n\n用户问题：{user_message}"},
        ]
```

**步骤 3：Python SSE API**

# app/api/chat.py
from fastapi import APIRouter
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
import json

router = APIRouter()

class ChatGenerateRequest(BaseModel):
    company_code: str
    company_name: str
    industry: str
    message: str
    history: list[dict] = []
    rag_context: list[dict] = []

@router.post("/ai/v1/chat/generate")
async def generate(req: ChatGenerateRequest):
    # 构建消息
    system_prompt = SYSTEM_PROMPT.format(
        company_name=req.company_name,
        company_code=req.company_code,
        industry=req.industry,
        financial_summary="(从数据库获取最新财报摘要)"
    )
    messages = llm_service.build_messages(system_prompt, req.message, req.rag_context)

    async def event_stream():
        try:
            yield f"data: {json.dumps({'type': 'thinking', 'content': '正在分析...'})}\n\n"
            async for token in llm_service.chat_stream(messages):
                yield f"data: {json.dumps({'type': 'token', 'content': token})}\n\n"
            # 返回引用
            refs = [{"source": c["source"], "snippet": c["content"][:100]} for c in req.rag_context]
            yield f"data: {json.dumps({'type': 'refs', 'refs': refs})}\n\n"
            yield f"data: {json.dumps({'type': 'done'})}\n\n"
        except Exception as e:
            yield f"data: {json.dumps({'type': 'error', 'content': '抱歉，生成回答时出错，请稍后重试。'})}\n\n"

    return StreamingResponse(event_stream(), media_type="text/event-stream")

**步骤 4：Java ChatService 编排**

```java
@Service
public class ChatService {

    public SseEmitter sendMessage(String companyCode, String message, String sessionId) {
        SseEmitter emitter = new SseEmitter(60000L);

        CompletableFuture.runAsync(() -> {
            try {
                // 1. RAG 检索
                RagSearchResponse ragResult = ragClient.search(message, companyCode, 5);

                // 2. 获取公司信息
                Company company = companyRepo.findByCode(companyCode);

                // 3. 调用 Python LLM
                ChatGenerateRequest req = ChatGenerateRequest.builder()
                    .companyCode(companyCode)
                    .companyName(company.getName())
                    .industry(company.getIndustry())
                    .message(message)
                    .ragContext(ragResult.getData())
                    .build();

                // 4. 转发 Python SSE 到前端
                pythonChatClient.generateStream(req, (token) -> {
                    try {
                        emitter.send(SseEmitter.event().data(token));
                    } catch (IOException e) { emitter.completeWithError(e); }
                });

                emitter.complete();
            } catch (Exception e) {
                // 降级：规则匹配回复
                String fallback = ruleBasedReply(message, companyCode);
                try {
                    emitter.send(SseEmitter.event().data("{\"type\":\"token\",\"content\":\"" + fallback + "\"}"));
                    emitter.complete();
                } catch (IOException ex) { emitter.completeWithError(ex); }
            }
        });

        return emitter;
    }

    private String ruleBasedReply(String message, String companyCode) {
        // 关键词匹配兜底（与 proto 类似逻辑）
        if (message.contains("盈利") || message.contains("赚钱")) return "根据最新财报...";
        if (message.contains("风险")) return "当前识别的主要风险包括...";
        return "抱歉，AI 服务暂时不可用，请稍后再试。";
    }
}
```

**步骤 5：前端 SSE 消费**

```typescript
// composables/useChat.ts
async function sendMessage(text: string) {
  store.addMessage('user', text)
  store.isLoading = true

  const response = await fetch('/api/v1/chat/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ companyCode: currentCode.value, message: text, sessionId: store.sessionId })
  })

  const reader = response.body!.getReader()
  const decoder = new TextDecoder()
  let assistantContent = ''

  while (true) {
    const { done, value } = await reader.read()
    if (done) break
    const chunk = decoder.decode(value)
    const lines = chunk.split('\n')
    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = JSON.parse(line.slice(6))
        if (data.type === 'token') {
          assistantContent += data.content
          // 更新最后一条助手消息（打字机效果）
          updateLastAssistantMessage(assistantContent)
        }
      }
    }
  }
  store.isLoading = false
}
```
