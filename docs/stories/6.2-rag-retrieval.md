# Story 6.2 — RAG 检索增强

> **Epic**: chat (智能问答) | **优先级**: P2 | **阶段**: Phase 3 AI 增强
> **依赖**: Story 6.1 | **预估工时**: 3d

---

## 一、PRD（产品需求文档）

### 1.1 背景

LLM 直接回答财报问题可能产生幻觉（编造数据）。RAG（检索增强生成）先检索相关财报段落，再将检索结果作为上下文注入 LLM，大幅提升回答准确性。

### 1.2 需求描述

作为系统，我需要在回答用户问题前先检索相关财报段落，将检索结果作为上下文提供给 LLM，确保回答有据可查。

### 1.3 验收标准

| # | 验收条件 |
|---|---------|
| AC1 | 财报文本按章节/段落切分为 chunks（~500 字，overlap 50 字） |
| AC2 | 每个 chunk 生成 Embedding 向量并存入 ChromaDB |
| AC3 | 用户问题 → Embedding → Top-5 相似 chunks 检索 |
| AC4 | 检索延迟 < 500ms |
| AC5 | 检索结果包含 source 标注（如"2024年报-P12"） |
| AC6 | 混合检索：BM25（关键词） + 向量（语义），取交集/加权 |

```python
# app/services/document_loader.py
from langchain.text_splitter import RecursiveCharacterTextSplitter

class DocumentLoader:
    def __init__(self, chunk_size=500, chunk_overlap=50):
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=chunk_size,
            chunk_overlap=chunk_overlap,
            separators=["\n\n", "\n", "。", "；", "，", " ", ""]
        )

    def load_and_split(self, text: str, source: str) -> list[dict]:
        chunks = self.splitter.split_text(text)
        return [
            {
                "content": chunk,
                "source": source,
                "chunk_index": i,
                "id": f"{source}_chunk_{i}"
            }
            for i, chunk in enumerate(chunks)
        ]
```

**步骤 3：Embedding 服务**

```python
# app/services/embedding_service.py
from openai import OpenAI
import numpy as np

class EmbeddingService:
    def __init__(self, api_key: str, model="text-embedding-3-small"):
        self.client = OpenAI(api_key=api_key)
        self.model = model

    def embed(self, texts: list[str]) -> list[list[float]]:
        """批量向量化"""
        response = self.client.embeddings.create(
            model=self.model,
            input=texts
        )
        return [d.embedding for d in response.data]

    def embed_query(self, query: str) -> list[float]:
        """单个查询向量化"""
        return self.embed([query])[0]
```

**步骤 4：RAG 检索服务**

```python
# app/services/rag_service.py
import chromadb
from rank_bm25 import BM25Okapi
import jieba

class RagService:
    def __init__(self, embedding_service: EmbeddingService):
        self.chroma = chromadb.PersistentClient(path="./chroma_data")
        self.collection = self.chroma.get_or_create_collection("financial_reports")
        self.embedding_service = embedding_service

    def index_documents(self, docs: list[dict]):
        """索引文档到向量库"""
        texts = [d["content"] for d in docs]
        embeddings = self.embedding_service.embed(texts)
        self.collection.add(
            embeddings=embeddings,
            documents=texts,
            metadatas=[{"source": d["source"], "chunk_index": d["chunk_index"]} for d in docs],
            ids=[d["id"] for d in docs]
        )

    def search(self, query: str, company_code: str, top_k=5) -> list[dict]:
        """混合检索：向量 + BM25"""
        # 1. 向量检索
        query_embedding = self.embedding_service.embed_query(query)
        vector_results = self.collection.query(
            query_embeddings=[query_embedding],
            n_results=top_k * 2,
            where={"source": {"$contains": company_code}}
        )

        # 2. BM25 关键词检索
        corpus = self._get_company_docs(company_code)
        tokenized_corpus = [list(jieba.cut(doc)) for doc in corpus]
        bm25 = BM25Okapi(tokenized_corpus)
        tokenized_query = list(jieba.cut(query))
        bm25_scores = bm25.get_scores(tokenized_query)

        # 3. 融合排序 (RRF: Reciprocal Rank Fusion)
        return self._rrf_fusion(vector_results, bm25_scores, top_k)

    def _rrf_fusion(self, vector_results, bm25_scores, top_k, k=60):
        """Reciprocal Rank Fusion 融合排序"""
        scores = {}
        # 向量排名分数
        for rank, (doc_id, doc, meta) in enumerate(zip(
            vector_results["ids"][0], vector_results["documents"][0],
            vector_results["metadatas"][0])):
            scores[doc_id] = {"score": 1.0 / (k + rank + 1), "content": doc, "source": meta["source"]}
        # BM25 分数归一化并融合
        max_bm25 = max(bm25_scores) if max(bm25_scores) > 0 else 1
        for i, s in enumerate(bm25_scores):
            doc_id = f"chunk_{i}"
            bm25_norm = s / max_bm25
            if doc_id in scores:
                scores[doc_id]["score"] += 0.3 * bm25_norm  # BM25 权重 0.3
        # 排序取 top_k
        sorted_items = sorted(scores.items(), key=lambda x: x[1]["score"], reverse=True)[:top_k]
        return [{"id": k, **v} for k, v in sorted_items]
```

**步骤 5：Python API 端点**

```python
# app/api/rag.py
from fastapi import APIRouter
from pydantic import BaseModel

router = APIRouter()

class SearchRequest(BaseModel):
    query: str
    company_code: str
    top_k: int = 5

@router.post("/ai/v1/rag/search")
async def search(request: SearchRequest):
    results = rag_service.search(request.query, request.company_code, request.top_k)
    return {"code": 0, "data": results}
```

**步骤 6：Java 调用端**

```java
// backend/.../client/RagClient.java
@Service
public class RagClient {
    private final RestTemplate restTemplate;

    public RagClient(@Value("${ai-engine.url}") String aiEngineUrl) {
        this.restTemplate = new RestTemplate();
        this.restTemplate.setUriTemplateHandler(new DefaultUriBuilderFactory(aiEngineUrl));
    }

    public RagSearchResponse search(String query, String companyCode, int topK) {
        RagSearchRequest req = new RagSearchRequest(query, companyCode, topK);
        return restTemplate.postForObject("/ai/v1/rag/search", req, RagSearchResponse.class);
    }
}
```
