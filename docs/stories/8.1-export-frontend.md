## 一、PRD（产品需求文档）

### 1.1 背景

用户分析完财报后，需要将报告导出为文件用于存档、分享或汇报。系统需支持 PNG/PDF/Word/Excel 四种格式。

### 1.2 需求描述

作为用户，我可以将当前分析报告导出为 PNG/PDF/Word/Excel 格式。

### 1.3 验收标准

| # | 验收条件 |
|---|---------|
| AC1 | Navbar 中显示"导出"按钮，点击展开格式选择菜单 |
| AC2 | 选择 PNG → 浏览器下载 PNG 截图 |
| AC3 | 选择 PDF → 浏览器下载 PDF 文件（A4 分页） |
| AC4 | 选择 Word → 浏览器下载 .docx 文件 |
| AC5 | 选择 Excel → 浏览器下载 .xlsx 文件 |
| AC6 | 文件名格式：`{公司名}_财报分析报告_{日期}.{格式}` |
| AC7 | 导出内容包含免责声明 |

## 二、开发方案

### 2.1 技术方案

| 格式 | 实现方式 | 依赖 |
|------|---------|------|
| PNG | `html2canvas` 截图 | html2canvas |
| PDF | `html2canvas` → `jsPDF` | html2canvas + jspdf |
| Word | 构建 HTML → Blob 下载 | 无需额外依赖 |
| Excel | 提取表格 → `exceljs` 生成 | exceljs |

### 2.2 关键文件

| 文件 | 说明 |
|------|------|
| `frontend/src/composables/useExport.ts` | 导出逻辑 |
| `frontend/src/components/ExportMenu.vue` | 导出格式选择菜单 |
| `frontend/src/components/Navbar.vue` | 集成导出按钮 |

**步骤 2：实现 useExport composable**

```typescript
// composables/useExport.ts
import html2canvas from 'html2canvas'
import jsPDF from 'jspdf'
import ExcelJS from 'exceljs'

export function useExport(companyName: Ref<string>) {
  const exporting = ref(false)
  const targetEl = ref<HTMLElement | null>(null)

  function filename(format: string): string {
    const date = new Date().toISOString().slice(0, 10)
    return `${companyName.value}_财报分析报告_${date}.${format}`
  }

  // 导出前附加免责声明
  function appendDisclaimer() {
    const el = document.createElement('div')
    el.className = 'export-disclaimer'
    el.innerHTML = '<p style="color:#92400e;border:1px dashed #f59e0b;padding:12px;margin-top:16px;font-size:12px">⚠️ 免责声明：本报告由 SmartReport 自动生成，仅供学习参考，不构成任何投资建议。</p>'
    targetEl.value?.appendChild(el)
    return el
  }

  async function exportPNG() {
    exporting.value = true
    const disc = appendDisclaimer()
    await new Promise(r => setTimeout(r, 100)) // 等待 DOM 渲染
    const canvas = await html2canvas(targetEl.value!, { scale: 2, useCORS: true })
    canvas.toBlob(blob => {
      const url = URL.createObjectURL(blob!)
      const a = document.createElement('a')
      a.href = url; a.download = filename('png')
      a.click(); URL.revokeObjectURL(url)
    })
    disc.remove()
    exporting.value = false
  }

  async function exportPDF() {
    exporting.value = true
    const disc = appendDisclaimer()
    await new Promise(r => setTimeout(r, 100))

    const canvas = await html2canvas(targetEl.value!, { scale: 2 })
    const imgData = canvas.toDataURL('image/png')

    const pdf = new jsPDF('p', 'mm', 'a4')
    const pageWidth = pdf.internal.pageSize.getWidth()
    const pageHeight = pdf.internal.pageSize.getHeight()
    const imgHeight = (canvas.height * pageWidth) / canvas.width

    let heightLeft = imgHeight
    let position = 0
    pdf.addImage(imgData, 'PNG', 0, position, pageWidth, imgHeight)
    heightLeft -= pageHeight

    while (heightLeft > 0) {
      position = -(imgHeight - heightLeft)
      pdf.addPage()
      pdf.addImage(imgData, 'PNG', 0, position, pageWidth, imgHeight)
      heightLeft -= pageHeight
    }

    pdf.save(filename('pdf'))
    disc.remove()
    exporting.value = false
  }

  async function exportWord() {
    const disc = appendDisclaimer()
    const html = targetEl.value!.innerHTML
    const blob = new Blob(['\ufeff' + html], { type: 'application/msword' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url; a.download = filename('doc')
    a.click(); URL.revokeObjectURL(url)
    disc.remove()
  }

  async function exportExcel() {
    const workbook = new ExcelJS.Workbook()
    const sheet = workbook.addWorksheet('财报数据')

    // 提取表格数据（从 DataTable 组件获取）
    const tableData = getTableData()  // 从 store 或 DOM 提取
    sheet.addRows(tableData)

    const buffer = await workbook.xlsx.writeBuffer()
    const blob = new Blob([buffer], { type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url; a.download = filename('xlsx')
    a.click(); URL.revokeObjectURL(url)
  }

  return { exporting, targetEl, exportPNG, exportPDF, exportWord, exportExcel }
}
```
