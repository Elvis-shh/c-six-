## 一、PRD（产品需求文档）

### 1.1 背景

OCR 提取的文本是非结构化的，需要从中识别出营收、净利润、现金流等关键财务指标，并结构化入库。

### 1.2 需求描述

作为系统，我需要从非结构化财报文本中提取关键财务指标，对低置信度字段提供人工确认界面。

### 1.3 验收标准

| # | 验收条件 |
|---|---------|
| AC1 | 从财报文本中提取：营收、净利润、总资产、负债、现金流、毛利率等核心指标 |
| AC2 | 每个提取字段附带置信度（0-1） |
| AC3 | 置信度 < 0.8 的字段标记为"需确认" |
| AC4 | 前端显示提取结果预览，用户可手动修正 |
| AC5 | 修正后数据写入 `financial_indicators` 表 |

### 2.3 实施步骤

**步骤 1：正则提取器**

```python
# app/services/regex_extractor.py
import re

class RegexExtractor:
    PATTERNS = {
        "revenue": [
            r"营业收入[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
            r"营业总收入[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
            r"实现营收[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
        ],
        "profit": [
            r"归[属母]?净利润[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
            r"净利润[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
        ],
        "totalAssets": [
            r"总资产[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
            r"资产总计[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
        ],
        "totalLiabilities": [
            r"总负债[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
            r"负债合计[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
        ],
        "cashFlow": [
            r"经营.*现金流[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
            r"经营活动.*现金[：:\s]*([\d,]+\.?\d*)\s*[万亿]元",
        ],
    }

    def extract(self, text: str) -> dict:
        results = {}
        for key, patterns in self.PATTERNS.items():
            for pattern in patterns:
                match = re.search(pattern, text)
                if match:
                    raw = match.group(1).replace(",", "")
                    value = float(raw)
                    # 如果是"万"元，转换为亿
                    if "万" in match.group(0) and "亿" not in match.group(0):
                        value /= 10000
                    results[key] = {
                        "value": value,
                        "confidence": 0.95,
                        "method": "regex",
                        "matched_text": match.group(0),
                    }
                    break
        return results
```

**步骤 2：LLM 辅助提取**

```python
# app/services/ner_service.py
class NerService:
    def __init__(self, llm_service: LLMService, regex_extractor: RegexExtractor):
        self.llm = llm_service
        self.regex = regex_extractor

    async def extract(self, text: str) -> dict:
        # 1. 先尝试正则提取
        regex_results = self.regex.extract(text)

        # 2. 未匹配到的指标，用 LLM 提取
        missing_keys = [k for k in self.regex.PATTERNS if k not in regex_results]
        if missing_keys:
            llm_results = await self._llm_extract(text, missing_keys)
            regex_results.update(llm_results)

        return regex_results

    async def _llm_extract(self, text: str, keys: list[str]) -> dict:
        prompt = f"""从以下财报文本中提取财务指标，返回 JSON 格式。

指标列表：{', '.join(keys)}

文本：
{text[:8000]}  # 限制长度

返回格式：
{{"revenue": {{"value": 1748, "unit": "亿"}}, "profit": {{"value": 862, "unit": "亿"}}}}
如果没有找到某指标，值设为 null。
"""
        response = await self.llm.chat([{"role": "user", "content": prompt}])
        try:
            data = json.loads(response)
            for key in data:
                if data[key] and data[key].get("value") is not None:
                    data[key]["confidence"] = 0.75  # LLM 提取置信度较低
                    data[key]["method"] = "llm"
            return data
        except json.JSONDecodeError:
            return {}
```

**步骤 4：Java 接收并入库**

```java
@Service
public class ExtractionService {

    public void saveExtractedData(String taskId, Map<String, ExtractedIndicator> data,
                                   String companyCode, int year) {
        FinancialReport report = reportService.getOrCreateReport(companyCode, year, "upload");

        for (var entry : data.entrySet()) {
            ExtractedIndicator ext = entry.getValue();
            if (ext.getValue() == null) continue;

            FinancialIndicator indicator = FinancialIndicator.builder()
                .reportId(report.getId())
                .indicatorKey(entry.getKey())
                .value(BigDecimal.valueOf(ext.getValue()))
                .build();
            indicatorRepo.save(indicator);
        }

        // 更新任务状态
        taskRepo.updateStatus(taskId, "completed");
    }
}
```
